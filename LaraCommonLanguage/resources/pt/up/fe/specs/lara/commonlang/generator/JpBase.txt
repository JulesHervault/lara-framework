/**
 * Base class for all LARA join points.
 *
 * @class
 */
var JoinPoint = function(astNode) {
	this.astNode = astNode;
};

JoinPoint.prototype._is_common_jp = true;

JoinPoint.prototype.instanceOf = function(joinPointType) {
	return joinPointType === this.joinPointType();
}

JoinPoint.prototype.joinPointType = function() {
	return "joinpoint";
}
